class LinearRegression(object):
    '''ÂàùÂßãÂåñ'''
    def __init__(self,b,lam):
        self.b=b        #ÂÅèÂ∑ÆÈ°π
        self.lam=lam    #Â≤≠ÂõûÂΩíÊâÄË¶ÅÁî®Âà∞ÁöÑÂ≤≠Á≥ªÊï∞
    '''Ê±ÇËß£Èó≠ÂêàÂΩ¢Âºè'''
    def fit_OLS(self,X,Y):  	#ÊúÄÂ∞è‰∫å‰πòÊ≥ïÁöÑÈó≠ÂêàÂΩ¢Âºè
        self.w=np.dot(np.dot(np.linalg.inv(np.dot(X.T,X)),X.T),Y)
    def fit_Ridge(self,X,Y):    #Â≤≠ÂõûÂΩíÁöÑÈó≠ÂêàÂΩ¢Âºè
        self.w=np.dot(np.dot(np.linalg.inv(np.dot(X.T,X)+self.lam*np.eye(len(X[0]))),X.T),Y)
    '''È¢ÑÊµãÁªìÊûú'''
    def predict(self,X_test):
        return np.dot(X_test,self.w)
