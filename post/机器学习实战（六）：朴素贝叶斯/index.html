class Naive_Bayes(object):
    def __init__(self,l):
        self.l=l    #Âπ≥ÊªëÂèÇÊï∞
    '''Ê®°ÂûãËÆ≠ÁªÉ'''
    def fit(self,X_train, y_train):
        '''ÁªüËÆ°Êï∞ÁõÆ'''
        numTrainDocs = len(X_train) #ËÆ°ÁÆóËÆ≠ÁªÉÁîµÂ≠êÈÇÆ‰ª∂Êï∞ÁõÆ
        d = len(X_train[0])  #ËÆ°ÁÆóÁîµÂ≠êÈÇÆ‰ª∂ÁöÑËØçÊù°Êï∞
        '''ËÆ°ÁÆóP(y=c)'''
        pAbusive = sum(y_train)/float(numTrainDocs)#ÊñáÊ°£Â±û‰∫éÂûÉÂúæÈÇÆ‰ª∂Á±ªÁöÑÁöÑÊ¶ÇÁéáP(y=spam)=œÄc
        '''ÂàùÂßãÂåñ'''
        p0Num = np.ones(d)
        p1Num = np.ones(d)
        p0Denom = self.l*d   #Áî®‰∫éËÆ∞ÂΩïhamÁ±ªÁöÑÁîµÂ≠êÈÇÆ‰ª∂ÂçïËØçÊÄªÊï∞+l
        p1Denom = self.l*d   #Áî®‰∫éËÆ∞ÂΩïspamÁ±ªÁöÑÁîµÂ≠êÈÇÆ‰ª∂ÂçïËØçÊÄªÊï∞+l
        '''Êï∞ÊçÆÁªüËÆ°'''
        for i in range(numTrainDocs):   #ÈÅçÂéÜÊØè‰∏Ä‰∏™ÁîµÂ≠êÈÇÆ‰ª∂
            if y_train[i] == 1:         #ÊòØÂûÉÂúæÈÇÆ‰ª∂spam
                p1Num += X_train[i]     #Áî®‰∫éspamÁ±ªÁöÑÂêÑ‰∏™ÂçïËØçÊï∞Èáè
                p1Denom += sum(X_train[i])
            else:                       #ÊòØÊ≠£Â∏∏ÈÇÆ‰ª∂ham
                p0Num += X_train[i]     #Áî®‰∫éÁªüËÆ°hamÁ±ªÁöÑÂêÑ‰∏™ÂçïËØçÊï∞Èáè
                p0Denom += sum(X_train[i])
        #Áõ∏Èô§ËÆ°ÁÆóÊ¶ÇÁéáÂêëÈáè
        p1Vect = np.log(p1Num+self.l / p1Denom)
        p0Vect = np.log(p0Num+self.l / p0Denom)
        '''‰øùÁïôËÆ°ÁÆóÁöÑÊï∞ÊçÆ'''
        self.p1Vect=p1Vect
        self.p0Vect=p0Vect
        self.pAbusive=pAbusive
    '''ÁªìÊûúÈ¢ÑÊµã'''
    def predict(self,X_test):
        y_pred=[]
        for x in X_test:
            p1=np.dot(x,self.p1Vect)+np.log(self.pAbusive)
            p0=np.dot(x,self.p0Vect)+np.log(1-self.pAbusive)
            if p1>p0:
                y_pred.append(1)
            else:
                y_pred.append(0)
        return np.array(y_pred)
