    def __init__(self,data,target):
        '''ÂàùÂßãÂåñ‰∏Ä‰∏™Áî®‰∫éÁªüËÆ°Ê†áÁ≠æÊï∞ÈáèÁöÑÊ®°Êùø'''

            return target[0]            #ËøîÂõûÁõ∏ÂêåÁöÑÊ†áÁ≠æ

    '''ËÆ°ÁÆóÊï∞ÊçÆÈõÜDÁöÑÂü∫Â∞ºÁ≥ªÊï∞'''
    def Gini_D(self,target):  #Êàë‰ª¨‰ªÖÈúÄË¶ÅtargetÂç≥ÂèØËÆ°ÁÆóÊï∞ÊçÆÈõÜDÁöÑÂü∫Â∞ºÁ≥ªÊï∞

        for y in target:
            labelNum[y]+=1
        G=1.0   #Âü∫Â∞ºÁ≥ªÊï∞
        for y in labelNum.keys():
            G-=(labelNum[y]/dataNum)**2
        return G
    '''ËÆ°ÁÆóÁâπÂæÅAÁöÑÂü∫Â∞ºÁ≥ªÊï∞'''
    def Gini_D_A(self,A,data,target):
        dataNum=len(data)

        :param threshold: ÁâπÂæÅÈòàÂÄº

    def findBestSplit(self,A,data,target):
