class Perceptron(object):
    '''ÂàùÂßãÂåñÊ®°Âûã'''
    def __init__(self, learning_rate = 0.01, max_iter = 200):
        self.lr = learning_rate     #Â≠¶‰π†Áéá
        self.max_iter = max_iter    #ÊúÄÂ§ßËÆ≠ÁªÉËΩÆÊï∞
    '''Ê®°ÂûãËÆ≠ÁªÉ'''
    def fit(self,data,label):                   #Ê≥®ÊÑèdataÂíålabelÈÉΩ‰∏∫numpyÊï∞ÁªÑ
        self.w = np.array([0.]*data.shape[1])   #ÊùÉÈáçÂêëÈáè
        self.b = np.array([0.])                 #ÂÅèÂ∑ÆÈ°π
        iter=0                                  #ËÆ≠ÁªÉËΩÆÊï∞
        while True:
            m=0
            for i in range(data.shape[1]):
                x=data[i]
                y=label[i]
                if y*(np.dot(self.w,x)+self.b)<=0:#ÂàÜÁ±ªÈîôËØØÔºåËøõË°åÊõ¥Êñ∞
                    self.w=self.w+self.lr*y*x
                    self.b=self.b+self.lr*y	#Ê≥®ÊÑèÂú®ËøôÈáåÊàë‰ª¨‰πüÂØπÂÅèÂ∑ÆÈ°πËøõË°å‰∫ÜÊõ¥Êñ∞
                    m=m+1
            if m==0:    #Âú®ËÆ≠ÁªÉÈõÜ‰∏äÊ≤°ÊúâÈîôËØØ
                break
            iter+=1
            if iter>=self.max_iter: #ËææÂà∞ÊúÄÂ§ßËÆ≠ÁªÉËΩÆÊï∞
                break
    '''ËøõË°åÈ¢ÑÊµã'''
    def predict(self,data):
        y_pre = np.matmul(data,self.w) + self.b
        for i in range(len(y_pre)):
            if y_pre[i]>=0:
                y_pre[i]=1
            else:
                y_pre[i]=-1
        return y_pre
