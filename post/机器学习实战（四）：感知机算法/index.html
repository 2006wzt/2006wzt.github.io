class Perceptron(object):
    '''åˆå§‹åŒ–æ¨¡å‹'''
    def __init__(self, learning_rate = 0.01, max_iter = 200):
        self.lr = learning_rate     #å­¦ä¹ ç‡
        self.max_iter = max_iter    #æœ€å¤§è®­ç»ƒè½®æ•°
    '''æ¨¡å‹è®­ç»ƒ'''
    def fit(self,data,label):                   #æ³¨æ„dataå’Œlabeléƒ½ä¸ºnumpyæ•°ç»„
        self.w = np.array([0.]*data.shape[1])   #æƒé‡å‘é‡
        self.b = np.array([0.])                 #åå·®é¡¹
        iter=0                                  #è®­ç»ƒè½®æ•°
        while True:
            m=0
            for i in range(data.shape[1]):
                x=data[i]
                y=label[i]
                if y*(np.dot(self.w,x)+self.b)<=0:#åˆ†ç±»é”™è¯¯ï¼Œè¿›è¡Œæ›´æ–°
                    self.w=self.w+self.lr*y*x
                    self.b=self.b+self.lr*y	#æ³¨æ„åœ¨è¿™é‡Œæˆ‘ä»¬ä¹Ÿå¯¹åå·®é¡¹è¿›è¡Œäº†æ›´æ–°
                    m=m+1
            if m==0:    #åœ¨è®­ç»ƒé›†ä¸Šæ²¡æœ‰é”™è¯¯
                break
            iter+=1
            if iter>=self.max_iter: #è¾¾åˆ°æœ€å¤§è®­ç»ƒè½®æ•°
                break
    '''è¿›è¡Œé¢„æµ‹'''
    def predict(self,data):
        y_pre = np.matmul(data,self.w) + self.b
        for i in range(len(y_pre)):
            if y_pre[i]>=0:
                y_pre[i]=1
            else:
                y_pre[i]=-1
        return y_pre
